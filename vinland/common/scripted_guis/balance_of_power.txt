balance_of_power = {
	scope = character
	
	effect = {
		random_held_title = {
			limit = {
				has_variable = title_struggle
			}
			set_local_variable = {
				name = struggle_target
				value = var:title_struggle
			}
			save_scope_as = this_struggle
		}
		set_variable = {
			name = opposing_bop_title
			value = local_var:struggle_target
		}
		set_variable = {
			name = own_bop_title
			value = scope:this_struggle
		}
	}
	
	is_shown = {
		
	}
	
	is_valid = {
		any_held_title = {
			has_variable = title_struggle
		}
	}
}

bop_pos_inter_1 = {
	scope = character

	effect = {
		remove_reserved_gold = 100
		var:own_bop_title = {
			random_de_jure_county = {
				limit = {
					OR = {
						top_liege = ROOT
						holder = ROOT
					}
				}
				change_development_level = 1
			}
			change_variable = {
				name = popular_support
				add = 2
			}
			custom_tooltip = BOP_ADD_2_POPULAR_SUPPORT_TT
		}
	}
	
	is_valid = {
		reserved_gold >= 100
	}
}

bop_pos_inter_5 = {
	scope = character

	effect = {
		remove_reserved_gold = 25
		add_prestige = -75
		var:own_bop_title = {
			random_de_jure_county = {
				limit = {
					OR = {
						top_liege = ROOT
						holder = ROOT
					}
				}
				culture = {
					change_cultural_acceptance = {
						target = ROOT.culture
						value = 1.5
						desc = BOP_CULTURAL_PRACTICES_RESPECTED
					}
					save_scope_as = other_culture
				}
				ROOT.culture = {
					change_cultural_acceptance = {
						target = scope:other_culture
						value = 1.5
						desc = BOP_CULTURAL_PRACTICES_RESPECTED
					}
				}
			}
			change_variable = {
				name = popular_support
				add = 2
			}
			custom_tooltip = BOP_ADD_2_POPULAR_SUPPORT_TT
		}
		add_stress = 10
	}
	
	is_valid = {
		reserved_gold >= 25
		prestige >= 75
		var:own_bop_title = {
			custom_tooltip = {
				text = BOP_NO_DIF_CULTURES
				any_de_jure_county = {
					OR = {
						top_liege = ROOT
						holder = ROOT
					}
					NOT = { culture = ROOT.culture }
				}
			}
		}
	}
}
